33template <class QType> std::string getShortUnitName(QType q) {
34 const std::unordered_map<std::type_index, std::unordered_map<double, const char *>> shortNameMap =
35 {{typeid(meter),
36 {
37 {meter.get_value(), "m"},
38 {decimeter.get_value(), "dm"},
39 {centimeter.get_value(), "cm"},
40 {millimeter.get_value(), "mm"},
41 {kilometer.get_value(), "km"},
42 {inch.get_value(), "in"},
43 {foot.get_value(), "ft"},
44 {yard.get_value(), "yd"},
45 {mile.get_value(), "mi"},
46 {tile.get_value(), "tile"},
47 }},
48 {typeid(degree), {{degree.get_value(), "deg"}, {radian.get_value(), "rad"}}}};
49
50 try {
51 return shortNameMap.at(typeid(q)).at(q.get_value());
52 } catch (const std::out_of_range &e) {
53 throw std::domain_error(
54 "You have requested the shortname of an unknown unit somewhere (likely odometry strings). "
55 "Shortname for provided unit is unspecified. You can override this function to add more "
56 "names or manually specify the name instead.");
57 }
58}
